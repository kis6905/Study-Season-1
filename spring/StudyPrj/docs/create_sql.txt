CREATE TABLE USER (
  USER_SEQ INT PRIMARY KEY,
  NAME VARCHAR(255),
  ADDRESS VARCHAR(255),
  TEAM_SEQ INT
);

CREATE TABLE TEAM (
  TEAM_SEQ INT PRIMARY KEY,
  NAME VARCHAR(255)
);

CREATE TABLE ROLE (
  ROLE_SEQ INT PRIMARY KEY,
  NAME VARCHAR(255)
);

CREATE TABLE USER_ROLE (
  USER_SEQ INT,
  ROLE_SEQ INT,
  PRIMARY KEY(USER_SEQ, ROLE_SEQ)
);
